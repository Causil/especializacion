    }{
    \ifnumcomp{#2}{>}{2}{
      \ifstrequal{#1}{alpha}{$\alpha^{#2}(#3)$}{$#1^{(#2)}(#3)$}
    }{
      \ifstrequal{#2}{2}{$#1^{\prime\prime}(#3)$}{$#1^{\prime}(#3)$}
    }
  }
  \hspace{-4.7mm}
  }
